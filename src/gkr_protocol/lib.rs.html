26/// GKR protocol error type.

37/// The state of the Verifier.
38pub struct Verifier<F: FftField> {
39    /// $r_0, r_1,..., r_n$.
40    r: Vec<Vec<F>>,
41
42    /// $m$.
43    m: Vec<F>,
44
45    /// Circuit
46    circuit: Circuit,
47
48    /// State of the verifier.
49    state: VerifierState<F>,
50}

52/// The inner state of the [`Verifier`].
53enum VerifierState<F: FftField> {
54    Empty,
55    RunningSumCheck {
56        /// $b$ and $c$.
57        bc: Vec<F>,
58
59        /// Verifier.
60        verifier: Box<SumCheckVerifier<F, W<F>>>,
61
62        /// $add_i$
63        add_i: DenseMultilinearExtension<F>,
64
65        /// $mul_i$
66        mul_i: DenseMultilinearExtension<F>,
67    },
68}

70impl<F: FftField> Verifier<F> {
71    /// Create a new `Verifier` with the claim of the `Prover`.
72    ///
73    /// At the start of the protocol picks a random
74    /// $r_0 \in \mathbb{F}^{k_0}$
75    /// and sets
76    /// $m_0 \leftarrow \tilde{D}(r_0)$.
77    /// The remainder of the protocol is devoted to confirming
78    /// that
79    /// $m_0 = \tilde{W}_0(r_0)$
80    pub fn new(circuit: Circuit) -> Self {
81        Self {
82            r: vec![],
83            m: vec![],
84            circuit,
85            state: VerifierState::Empty,
86        }
87    }
88
89    fn start_round(&mut self, c_1: F, round: usize, num_vars: usize) -> Result<VerifierMessage<F>> {
90        let add_i = self.circuit.add_i_ext(self.r.last().unwrap(), round);
91        let mul_i = self.circuit.mul_i_ext(self.r.last().unwrap(), round);
92        let mut verifier = SumCheckVerifier::new(num_vars, None);
93        verifier.set_c_1(c_1);
94
95        let bc = vec![];
96
97        self.state = VerifierState::RunningSumCheck {
98            bc,
99            verifier: Box::new(verifier),
100            add_i,
101            mul_i,
102        };
103
104        Ok(VerifierMessage::RoundStarted(round))
105    }
106
107    /// Final random point in the Sum-Check protocol.
108    pub fn final_random_point<R: Rng>(&mut self, rng: &mut R) -> Result<VerifierMessage<F>> {
109        if let VerifierState::RunningSumCheck { bc, .. } = &mut self.state {
110            let final_point = F::rand(rng);
111            bc.push(final_point);
112
113            Ok(VerifierMessage::SumCheckRoundResult {
114                res: SumCheckVerifierRoundResult::JthRound(final_point),
115            })
116        } else {
117            Err(Error::WrongVerifierState)
118        }
119    }
120
121    fn sum_check_step<R: Rng>(
122        &mut self,
123        message: univariate::SparsePolynomial<F>,
124        rng: &mut R,
125    ) -> Result<VerifierMessage<F>> {
126        if let VerifierState::RunningSumCheck { bc, verifier, .. } = &mut self.state {
127            let res = verifier.round(message, rng).unwrap();
128
129            if let SumCheckVerifierRoundResult::JthRound(point) = res {
130                bc.push(point);
131            }
132
133            Ok(VerifierMessage::SumCheckRoundResult { res })
134        } else {
135            Err(Error::WrongVerifierState)
136        }
137    }
138
139    fn final_round_message<R: Rng>(
140        &mut self,
141        p: univariate::SparsePolynomial<F>,
142        q: univariate::SparsePolynomial<F>,
143        rng: &mut R,
144    ) -> Result<VerifierMessage<F>> {
145        if let VerifierState::RunningSumCheck {
146            bc, add_i, mul_i, ..
147        } = &self.state
148        {
149            /*
150             * TODO: check q degree
151             */
152            let q_0 = q.evaluate(&F::zero());
153            let q_1 = q.evaluate(&F::one());
154
155            let eval = add_i.evaluate(bc) * (q_0 + q_1) + mul_i.evaluate(bc) * q_0 * q_1;
156
157            assert_eq!(eval, p.evaluate(bc.last().unwrap()));
158
159            let r = F::rand(rng);
160            let (b, c) = bc.split_at(bc.len() / 2);
161
162            let line = line(b, c);
163
164            let r_next: Vec<F> = line.into_iter().map(|e| e.evaluate(&r)).collect();
165            let m_next = q.evaluate(&r);
166
167            self.r.push(r_next.clone());
168            self.m.push(m_next);
169
170            Ok(VerifierMessage::R { r: r_next })
171        } else {
172            Err(Error::WrongVerifierState)
173        }
174    }
175
176    /// Receive a message from [`Prover`].
177    pub fn receive_prover_msg<R: Rng>(
178        &mut self,
179        msg: ProverMessage<F>,
180        rng: &mut R,
181    ) -> Result<VerifierMessage<F>> {
182        match msg {
183            ProverMessage::SumCheckProverMessage { p } => self.sum_check_step(p, rng),
184            ProverMessage::StartSumCheck {
185                c_1,
186                round,
187                num_vars,
188            } => self.start_round(c_1, round, num_vars),
189            ProverMessage::FinalRoundMessage { p, q } => self.final_round_message(p, q, rng),
190            ProverMessage::Begin { circuit_outputs } => {
191                let num_output_vars = self.circuit.num_vars_at(0).unwrap();
192                let d = DenseMultilinearExtension::from_evaluations_slice(
193                    num_output_vars,
194                    &circuit_outputs,
195                );
196
197                let r_zero: Vec<_> = (0..num_output_vars).map(|_| F::rand(rng)).collect();
198
199                let m_zero = d.evaluate(&r_zero);
200
201                self.r = vec![r_zero.clone()];
202                self.m = vec![m_zero];
203
204                Ok(VerifierMessage::R { r: r_zero })
205            }
206        }
207    }
208
209    /// Perform the final check of the input.
210    pub fn check_input(&self, input: &[F]) -> bool {
211        let w = DenseMultilinearExtension::from_evaluations_slice(
212            (f64::from(input.len() as u32)).log2() as usize,
213            input,
214        );
215
216        &w.evaluate(self.r.last().unwrap()) == self.m.last().unwrap()
217    }
218}

220/// Messages emitted by the [`Verifier`].
221#[derive(Debug)]
222pub enum VerifierMessage<F: Field> {
223    /// A result of running a step in the current sum check protocol.
224    SumCheckRoundResult {
225        /// Result of a Sum-Check round.
226        res: SumCheckVerifierRoundResult<F>,
227    },
228
229    /// The first round has completed.
230    FirstRound,
231
232    /// The j-th round has started.
233    RoundStarted(usize),
234
235    /// Sends out the $r_i$ to be used by the [`Prover`].
236    R {
237        /// $r_i$
238        r: Vec<F>,
239    },
240}

242/// Messages emitted by the [`Prover`].
243#[derive(Debug, PartialEq, Eq)]
244pub enum ProverMessage<F: Field> {
245    /// [`Prover`] begins the protocol by the claim about the outputs.
246    Begin {
247        /// Claimed outputs
248        circuit_outputs: Vec<F>,
249    },
250    /// A step of the current sum-check protocol.
251    SumCheckProverMessage {
252        /// A polynomial sent at each step of Sum-Check.
253        p: univariate::SparsePolynomial<F>,
254    },
255    /// In the final the restriction polynomial $q$ is added.
256    FinalRoundMessage {
257        /// A polynomial sent at each step of Sum-Check.
258        p: univariate::SparsePolynomial<F>,
259
260        /// Sends a univariate polynomial $q$ of degree at most
261        /// k_{i+1} claimed to equal $\tilde{W}_{i+1}$ to $l$
262        q: univariate::SparsePolynomial<F>,
263    },
264    /// Instruct the [`Verifier`] to start a Sum-Check protocol for some round.
265    StartSumCheck {
266        /// A $c_1$ from Sum-Check protocol.
267        c_1: F,
268
269        /// At which round of GKR this Sum-Check is being started.
270        round: usize,
271
272        /// A number of variables.
273        num_vars: usize,
274    },
275}

277/// Compute a line
278pub fn line<F: Field>(b: &[F], c: &[F]) -> Vec<univariate::SparsePolynomial<F>> {
279    iter::zip(b, c)
280        .map(|(b, c)| {
281            univariate::SparsePolynomial::from_coefficients_slice(&[(0, *b), (1, *c - b)])
282        })
283        .collect()
284}
285
286/// Restrict a polynomial to a given line.
287///
288/// Given two points $b,c \in \mathbb{F}^{\log n}$ create a line $l$
289/// such that $l(0) = b$ and $l(1) = c$ and restrict an MLE $\tilde{W}$
290/// to this line outputting a univariate polynomial.
291pub fn restrict_poly<F: Field, M: MultilinearExtension<F>>(
292    b: &[F],
293    c: &[F],
294    mle: &M,
295) -> univariate::SparsePolynomial<F> {
296    let k: Vec<_> = iter::zip(b, c).map(|(b, c)| *c - b).collect();
297
298    let evaluations = mle.to_evaluations();
299    let num_vars = mle.num_vars();
300
301    let mut res = univariate::SparsePolynomial::zero();
302
303    for (i, evaluation) in evaluations.iter().enumerate() {
304        let mut p = univariate::SparsePolynomial::from_coefficients_vec(vec![(0, *evaluation)]);
305        for bit in 0..num_vars {
306            let mut b =
307                univariate::SparsePolynomial::from_coefficients_vec(vec![(0, b[bit]), (1, k[bit])]);
308
309            if i & (1 << bit) == 0 {
310                b = (&univariate::DensePolynomial::from_coefficients_vec(vec![F::one()]) - &b)
311                    .into();
312            }
313
314            p = p.mul(&b);
315        }
316
317        res += &p;
318    }
319
320    res
321}

323/// The state of the Prover.
324pub struct Prover<F: FftField> {
325    /// Current round of the protocol.
326    i: usize,
327
328    /// The circuit.
329    circuit: Circuit,
330
331    /// Evaluations of the circuit on a given input.
332    evaluation: CircuitEvaluation<F>,
333
334    /// A Sum-Check protocol prover.
335    prover: Option<SumCheckProver<F, W<F>>>,
336
337    /// Current $\tilde{W}_{i+1}$.
338    w: DenseMultilinearExtension<F>,
339
340    /// Random points collected through a single Sum-Check protocol run.
341    r: Vec<F>,
342}
343
344impl<F: FftField> Prover<F> {
345    /// Create a new `Prover` state from a circuit and an evaluation.
346    pub fn new(circuit: Circuit, input: &[F]) -> Self {
347        let evaluation = circuit.evaluate(input);
348
349        Self {
350            i: 0,
351            circuit,
352            evaluation,
353            prover: None,
354            w: Default::default(),
355            r: vec![],
356        }
357    }
358
359    /// At the start of the protocol $P$ sends a function
360    /// $D: \lbrace 0, 1 \rbrace ^{k_0} \rightarrow \mathbb{F}$
361    /// claimed to equal $W_0$ (the function mapping output gate
362    /// labels to output values).
363    pub fn start_protocol(&self) -> ProverMessage<F> {
364        ProverMessage::Begin {
365            circuit_outputs: self.evaluation.layers.first().unwrap().clone(),
366        }
367    }
368
369    /// Create a Sum-Check prover for round $i$.
370    ///
371    /// At round $i$ a Sum-Check prover for polynomial
372    /// $f^{(i)}_{r_i}(b, c)$.
373    pub fn start_round(&mut self, i: usize, r_i: &[F]) -> ProverMessage<F> {
374        let num_vars_current = self.circuit.num_vars_at(i).unwrap();
375
376        let num_vars_next = self.circuit.num_vars_at(i + 1).unwrap();
377
378        let w_b = DenseMultilinearExtension::from_evaluations_slice(
379            num_vars_next,
380            &self.evaluation.layers[i + 1],
381        );
382
383        self.w = w_b.clone();
384        let w_c = w_b.clone();
385        let mut add_i = vec![];
386        let mut mult_i = vec![];
387
388        for c in 0..2usize.pow(num_vars_next as u32) {
389            for b in 0..2usize.pow(num_vars_next as u32) {
390                for a in 0..2usize.pow(num_vars_current as u32) {
391                    add_i.push(match self.circuit.add_i(i, a, b, c) {
392                        true => F::one(),
393                        false => F::zero(),
394                    });
395
396                    mult_i.push(match self.circuit.mul_i(i, a, b, c) {
397                        true => F::one(),
398                        false => F::zero(),
399                    });
400                }
401            }
402        }
403
404        let add_i = DenseMultilinearExtension::from_evaluations_vec(
405            num_vars_current + num_vars_next * 2,
406            add_i,
407        );
408        let mult_i = DenseMultilinearExtension::from_evaluations_vec(
409            num_vars_current + num_vars_next * 2,
410            mult_i,
411        );
412
413        let add_i = add_i.fix_variables(r_i);
414        let mult_i = mult_i.fix_variables(r_i);
415
416        let num_vars = add_i.num_vars();
417
418        assert_eq!(add_i.num_vars(), mult_i.num_vars());
419        assert_eq!(add_i.num_vars(), 2 * w_b.num_vars());
420
421        let w = W::new(add_i, mult_i, w_b, w_c);
422        self.i = i;
423
424        let prover = SumCheckProver::new(w);
425        let c_1 = prover.c_1();
426        self.prover = Some(prover);
427        self.r = vec![];
428
429        ProverMessage::StartSumCheck {
430            c_1,
431            round: i,
432            num_vars,
433        }
434    }
435
436    /// Perform a step of the Sum-Check protocol and provide a message for the [`Verifier`].
437    pub fn round_msg(&mut self, j: usize) -> ProverMessage<F> {
438        if j == 2 * self.circuit.num_vars_at(self.i + 1).unwrap() - 1 {
439            // The last round; do the polynomial restriction.
440            let (b, c) = self.r.split_at(self.r.len() / 2);
441
442            let q = restrict_poly(b, c, &self.w);
443
444            let p = self.prover.as_mut().unwrap().round(self.r[j - 1], j);
445            ProverMessage::FinalRoundMessage { p, q }
446        } else {
447            // Just a Sum-Check round
448            let point = if j == 0 { F::one() } else { self.r[j - 1] };
449
450            ProverMessage::SumCheckProverMessage {
451                p: self.prover.as_mut().unwrap().round(point, j),
452            }
453        }
454    }
455
456    /// Receive a message from the [`Verifier`].
457    pub fn receive_verifier_msg(&mut self, verifier_msg: VerifierMessage<F>) {
458        if let VerifierMessage::SumCheckRoundResult { res } = verifier_msg {
459            match res {
460                SumCheckVerifierRoundResult::JthRound(r_j) => {
461                    self.r.push(r_j);
462                }
463                SumCheckVerifierRoundResult::FinalRound(_) => panic!(),
464            }
465        }
466    }
467
468    /// Get the $c_1$ of the current Sum-Check prover.
469    pub fn c_1(&self) -> F {
470        self.prover.as_ref().unwrap().c_1()
471    }
472}

692            let verifier_msg = verifier.receive_prover_msg(prover_msg, rng).unwrap();
693
694            match verifier_msg {
695                VerifierMessage::R { r } => r_i = r,
696                _ => panic!("{:?}", verifier_msg),
697            }
698        }
699
700        assert!(verifier.check_input(&input));
701    }
