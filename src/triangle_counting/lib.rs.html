10/// A polynomial
11///
12/// $$
13/// g(X,Y,Z) = \tilde{f}_A(X,Y) \cdot \tilde{f}_A(Y,Z) \cdot \tilde{f}_A(X,Z)
14/// $$
15///
16/// that when used with Sum Check protocol yields an IP for computing a $6\Delta$
17/// number of triangles in a graph.
18///
19/// Holds three copies of a multilinear extension $\tilde{f}_A$ since fixing
20/// a variable in $g$ will lead to different multilinear equations.
21#[derive(Clone)]
22pub struct G<F: Field> {
23    f_a_1: DenseMultilinearExtension<F>,
24    f_a_2: DenseMultilinearExtension<F>,
25    f_a_3: DenseMultilinearExtension<F>,
26    var_len: usize,
27}
28
29impl<F: Field> G<F> {
30    /// Creates a new $3 \log n$-variate polynomial $g(X,Y,Z)$ from
31    /// a graph adjacency matrix.
32    pub fn new_adj_matrix<M>(num_vars: usize, matrix: M) -> Self
33    where
34        M: IntoIterator<Item = bool>,
35    {
36        let g = DenseMultilinearExtension::from_evaluations_vec(
37            num_vars,
38            matrix
39                .into_iter()
40                .map(|b| if b { F::one() } else { F::zero() })
41                .collect(),
42        );
43
44        let var_len = g.num_vars() / 2;
45        Self {
46            f_a_1: g.clone(),
47            f_a_2: g.clone(),
48            f_a_3: g,
49            var_len,
50        }
51    }
52
53    fn x_vars_num(&self) -> usize {
54        self.f_a_1.num_vars().saturating_sub(self.var_len)
55    }
56
57    fn y_vars_num(&self) -> usize {
58        self.f_a_2.num_vars().saturating_sub(self.var_len)
59    }
60
61    fn z_vars_num(&self) -> usize {
62        if self.f_a_3.num_vars() < self.var_len {
63            self.f_a_3.num_vars()
64        } else {
65            self.var_len
66        }
67    }
68}

70impl<F: FftField> SumCheckPolynomial<F> for G<F> {
71    fn evaluate(&self, point: &[F]) -> Option<F> {
72        let x_y_point = point
73            .get(..self.x_vars_num() + self.y_vars_num())
74            .unwrap_or(&[]);
75
76        let y_z_point = point.get(self.x_vars_num()..).unwrap_or(&[]);
77        let f_x_y_evaluation = self.f_a_1.evaluate(&x_y_point.into());
78        let f_y_z_evaluation = self.f_a_2.evaluate(&y_z_point.into());
79
80        let mut x_z = point.get(..self.x_vars_num()).unwrap_or(&[]).to_owned();
81
82        x_z.extend_from_slice(&point[self.x_vars_num() + self.y_vars_num()..]);
83
84        let f_x_z_evaluation = self.f_a_3.evaluate(&x_z);
85
86        Some(f_x_y_evaluation * f_x_z_evaluation * f_y_z_evaluation)
87    }
88
89    fn fix_variables(&self, partial_point: &[F]) -> Self {
90        let x_y_point = partial_point
91            .get(..cmp::min(self.x_vars_num() + self.y_vars_num(), partial_point.len()))
92            .unwrap_or(&[]);
93
94        let y_z_point = &partial_point.get(self.x_vars_num()..).unwrap_or(&[]);
95
96        let mut x_z_point = partial_point
97            .get(..cmp::min(self.x_vars_num(), partial_point.len()))
98            .unwrap_or(&[])
99            .to_owned();
100
101        x_z_point.extend_from_slice(
102            partial_point
103                .get(self.x_vars_num() + self.y_vars_num()..)
104                .unwrap_or(&[]),
105        );
106
107        let f_a_1 = self.f_a_1.fix_variables(x_y_point);
108        let f_a_2 = self.f_a_2.fix_variables(y_z_point);
109        let f_a_3 = self.f_a_3.fix_variables(&x_z_point);
110        let var_len = self.var_len;
111
112        Self {
113            f_a_1,
114            f_a_2,
115            f_a_3,
116            var_len,
117        }
118    }
119
120    fn to_univariate(&self) -> SparsePolynomial<F> {
121        let domain = GeneralEvaluationDomain::new(3).unwrap();
122
123        let evals = domain
124            .elements()
125            .map(|e| self.fix_variables(&[e]).to_evaluations().into_iter().sum())
126            .collect();
127
128        let evaluations = Evaluations::from_vec_and_domain(evals, domain);
129        let p = evaluations.interpolate();
130
131        p.into()
132    }
133
134    fn num_vars(&self) -> usize {
135        self.x_vars_num() + self.y_vars_num() + self.z_vars_num()
136    }
137
138    fn to_evaluations(&self) -> Vec<F> {
139        // combine the evaluations of separate multilinear
140        // extensions into a vector of evaluations of the
141        // whole polynomial
142        let f_a_1_evals = self.f_a_1.to_evaluations();
143        let f_a_2_evals = self.f_a_2.to_evaluations();
144        let f_a_3_evals = self.f_a_3.to_evaluations();
145
146        let mut res = vec![];
147        for x_idx in 0..2usize.pow(self.x_vars_num() as u32) {
148            for y_idx in 0..2usize.pow(self.y_vars_num() as u32) {
149                for z_idx in 0..2usize.pow(self.z_vars_num() as u32) {
150                    // indexing into evaluations in this order: x, y, z
151                    // and so when combining for example x and y
152                    // x always occupies the lower bits and y occupies the
153                    // higher bits in the index.
154
155                    let idx_1 = idx(y_idx, x_idx, self.x_vars_num());
156                    let idx_2 = idx(z_idx, y_idx, self.y_vars_num());
157                    let idx_3 = idx(z_idx, x_idx, self.x_vars_num());
158
159                    res.push(f_a_1_evals[idx_1] * f_a_2_evals[idx_2] * f_a_3_evals[idx_3])
160                }
161            }
162        }
163
164        res
165    }
166}
167
168/// Combine indices of two variables into one to be able
169/// to index into evaluations of polynomial.
170fn idx(i: usize, j: usize, num_vars: usize) -> usize {
171    (i << num_vars) | j
172}

183    struct AdjMatrix(Vec<Vec<bool>>);
184
185    impl AdjMatrix {
186        fn new<R: Rng>(n: usize, rng: &mut R) -> Self {
187            let mut res = Vec::with_capacity(n);
188
189            for _ in 0..n {
190                res.push(vec![false; n]);
191            }
192
193            for i in 0..n {
194                for j in 0..n {
195                    if i != j {
196                        let are_connected = rng.gen();
197
198                        res[i][j] = are_connected;
199                        res[j][i] = are_connected;
200                    }
201                }
202            }
203
204            Self(res)
205        }
206
207        fn triangle_count(&self) -> usize {
208            let mut count = 0;
209            let n = self.0.len();
210
211            for x in 0..n {
212                for y in 0..n {
213                    for z in 0..n {
214                        if self.0[x][y] & self.0[y][z] & self.0[x][z] {
215                            count += 1;
216                        }
217                    }
218                }
219            }
220
221            count / 6
222        }
223    }
